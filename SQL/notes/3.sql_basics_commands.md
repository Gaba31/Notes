```sql
 -- This is how we can create table

create table employee(
employee_id int primary key auto_increment,
first_name varchar(50) not null,
last_name varchar(50) not null,
hire_date DATE default (current_date()),
email varchar(50) unique,
phone_no varchar(50) unique,
salary decimal(10,2) check (salary>0.0),
employement_status enum('active','on_leave','terminated') default 'active',
created_at datetime default current_timestamp,
updated_at datetime default current_timestamp on update current_timestamp
);

create table department(
department_id int primary key auto_increment,
department_name varchar(100) not null,

updated_at timestamp default current_timestamp on update current_timestamp
);

-- How to update table
-- We have to add a new column with constraints 
-- But we cannot do directly like we can do with default values 
-- or we can just create a column then add values then apply constraint over it
-- like we are doing here

alter table employee 
add column emergency_contact varchar(100); 

-- How to rename the existing column in the table
alter table employee 
rename column emergency_contact to em_contact;

-- How to add constraint on the existing column
alter table employee
```
